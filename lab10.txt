 en varchar2(10);
 cursor ven_cursor is
 select v.vname from vendor v,part p,supply s where  s.vid=v.vid and p.pid=s.pid and p.pid=d ;
 begin
 open ven_cursor;
 loop
 fetch ven_cursor into en;
 exit when ven_cursor%notfound;
 dbms_output.put_line(' name is: '||en);
 end loop;
end;
